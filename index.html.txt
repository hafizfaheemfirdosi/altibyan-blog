<script>
document.addEventListener('DOMContentLoaded', function() {
  // Get all read more buttons
  const readMoreBtns = document.querySelectorAll('.read-more-btn');
  
  // Get all close buttons
  const closeBtns = document.querySelectorAll('.close-story');
  
  // Add click event to read more buttons
  readMoreBtns.forEach(btn => {
    btn.addEventListener('click', function() {
      // Hide all story contents first
      document.querySelectorAll('.story-content').forEach(content => {
        content.style.display = 'none';
      });
      
      // Show the selected story content
      const targetId = this.getAttribute('data-target');
      document.getElementById(targetId).style.display = 'block';
      
      // Scroll to the content
      document.getElementById(targetId).scrollIntoView({ behavior: 'smooth' });
    });
  });
  
  // Add click event to close buttons
  closeBtns.forEach(btn => {
    btn.addEventListener('click', function() {
      // Hide the parent story content
      this.closest('.story-content').style.display = 'none';
    });
  });
});
</script>
